<script>
    function hitungBMI() {
        let nama = document.getElementById("nama").value;
        let tinggi = parseFloat(document.getElementById("tinggi").value) / 100;
        let berat = parseFloat(document.getElementById("berat").value);

        if (!nama || !tinggi || !berat) {
            alert("Harap isi semua data!");
            return;
        }

        let bmi = (berat / (tinggi * tinggi)).toFixed(2);
        let kategori = "";

        if (bmi < 18.5) kategori = "Kurus";
        else if (bmi < 24.9) kategori = "Normal";
        else if (bmi < 29.9) kategori = "Berlebih";
        else kategori = "Obesitas";

        document.getElementById("result").style.display = "block";
        document.getElementById("result").innerHTML =
            `<b>${nama}</b><br>BMI Anda: <b>${bmi}</b><br>Status: <b>${kategori}</b>`;
    }

    function resetForm() {
        document.getElementById("nama").value = "";
        document.getElementById("usia").value = "";
        document.getElementById("gender").value = "";
        document.getElementById("tinggi").value = "";
        document.getElementById("berat").value = "";
        document.getElementById("result").style.display = "none";
    }
</script>
